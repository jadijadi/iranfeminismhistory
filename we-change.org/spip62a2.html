روناک و هانا و ياسر؛ تلاش برای رفع تبعيض‌های جنسيتی و ستم ملی، کاوه کرمانشاهی/خبرنامه گویا

چهار شنبه3 بهمن 1386


روناک صفازاده، هانا عبدی و ياسر گلی پس از آن‌که سه ماه را در بازداشت‌گاه اداره اطلاعات سنندج گذراندند اخيراً به زندان مرکزی اين شهر منتقل شدند و خانواده آنان توانستند با عزيزان‌شان ملاقات کنند هر چند اين ديدار باعث افزايش نگرانی خانواده‌ها نسبت به وضعيت نگهداری فرزندان‌شان شد به گونه‌ای که مادر ياسر می‌گويد: "زمانی که پسرم را ديدم قيافه‌ی صدام حسين پس از آن‌که از مخفی‌گاهش بيرون آمد در مقابل چشمانم ظاهر شد! موهای ژوليده و ريش انبوه ياسر نشان از قرار داشتن او تحت سخت‌ترين شرايط است طوری که حتی امکان شانه زدن موها و يا اصلاح صورتش را هم ندارد و شايد در اين مدت حتی يک بار هم اجازه استحمام به او داده نشده اما خوب با اين وجود پسرم مانند هميشه آرام و موقر بود و اعتماد به نفسش را از دست نداده بود..."*

وکلای سه جوان سنندجی نيز پس از گذشت بيش از سه ماه همچنان در انتظار تکميل مراحل تحقيق توسط نهادهای اطلاعاتی و قضايی هستند تا با تکميل تحقيقات و امکان دسترسی به پرونده و ديدار با موکلين خود مراحل حقوقی وکالت خويش و دفاع از متهمان را آغاز کنند و اين در حالی‌ست که هر از چند گاهی برخی از رسانه‌ها و خبرگزاری‌های دولتی اتهاماتی را از قول مقام‌های ‌امنيتی و قضايی مرتبط با پرونده روناک و هانا و ياسر مطرح می‌کنند مبنی بر وابستگی اين افراد به گروه‌های مسلح مخالف نظام که به گفته‌ی حکومت فعاليت‌های تروريستی انجام می‌دهند و نيز مشارکت در بمب‌گذاری‌های اخير شهر سنندج که هر چند هيچ گروهی (از جمله سازمانی که اين جوانان متهم به هم‌کاری با آن شده‌اند) مسئوليت انجام آن‌ها را بر عهده نگرفته اما با توجه به فعاليت گسترده گروه‌های تندرو اسلامی (سلفی‌ها) در کردستان بسياری بر اين باورند که همين گروه‌ها عوامل بمب‌گذاری‌ها مورد اشاره بوده‌اند. اما آن گونه که پيداست حاکميت چندان ميلی به تلاش در جهت کشف حقيقت و افشای هويت اصلی آمران و عاملان اين وقايع ندارد و ترجيح می‌دهد خود را با گروه‌های بنيادگرای اسلامی در کردستان که دارای قدرت و نفوذ چشم‌گيری نيز هستند درگير نکند و در عوض انگشت اتهام را به سوی جوانان بی‌پناهی نشانه گيرد و کسانی را تروريست معرفی نمايد که سابقه‌ی فعاليت‌های مدنی‌شان در قالب تشکل‌های قانونی چون انجمن زنان آذرمهر و اتحاديه دمکراتيک دانشجويان کُرد و هم‌راهی‌شان با جنبش‌های اجتماعی مانند جنبش زنان و جنبش دانشجويی و نيز حمايت‌شان از حرکت‌های مسالمت‌آميزی مثل کمپين يک ميليون امضاء برای تغيير قوانين تبعيض‌آميز نشان دهنده نوع فعاليت و شيوه عمل‌ و طرز فکرشان می‌باشد و بی‌اساس بودن اتهامات عنوان شده عليه آن‌ها را مشخص می‌کند.

نگين شيخ‌الاسلامی دبير انجمن آذرمهر که روناک و هانا از فعال‌ترين اعضای آن بوده و مادر ياسر نيز در گروه مادران اين تشکل عضويت دارد با تعجب از اتهامات نسبت داده شده به دوستانش می‌گويد: «چگونه ممکن است انسان‌هايی که ميزان ارزش و احترام‌شان به زندگی و حق حيات موجودات زنده به اندازه‌ای‌ست که از خوردن گوشت اجتناب می‌کنند و بر اين باورند که هر موجود زنده در جهان حق زيستن دارد و ما نبايد برای لذت نفسانی خويش آنان را از ادامه‌ی زندگی و حق حيات محروم کنيم تروريست باشند؟!
کسانی که از آغاز جوانی دغدغه‌ی کمک به مردم ديارشان و رفع مشکلات زنان و کودکان که در سرزمين ما آسيب‌پذيرترين قشر جامعه هستند را داشته‌اند، روستا به روستا برای زنان کلاس‌های آموزشی برگزار کرده‌اند و با هزينه‌ی شخصی و از حقوق اندک خود برای کودکان خانواده‌های کم‌درآمد کتاب ‌خريده‌اند، آنان که از خشونت عليه زنان و کودکان در اشکال مختلف از جمله سنت غلط ختنه‌ی دختران که هنوز هم در برخی مناطق دورافتاده روستايی در کردستان رايج است رنج ‌برده‌اند و تمام تلاش‌شان آگاه‌سازی زنان و حل اين مشکلات بوده چگونه ممکن است اين همه احساسات انسان‌دوستانه و اين همه فعاليت پی‌گير و مؤثر را رها کنند و به اعمال خشونت‌آميز يا به گفته‌ی آقايان تروريستی روی آورند؟!

آخر چگونه می‌توانند چنين القابی را به راحتی در مورد اين جوانان که پر از اميد و آرزو هستند به کار ببرند و اين اتهامات سنگين را به آنان نسبت دهند؟! برای من که از نزديک اين جوانان را می‌شناسم و از افکار و رفتار آن‌ها به خوبی آگاهم اتهامات نسبت داده شده به آنان به هيچ عنوان باور کردنی نيست حتی اگر به گفته‌ی آقايان اداره اطلاعات خودشان هم اعتراف کرده باشند.»

شيرين عبادی حقوق‌دان و برنده نوبل صلح که خود داوطلب وکالت روناک و هانا شده بود اما دادسرای انقلاب سنندج به بهانه‌ی عدم اتمام تحقيقات وکلات وی را نپذيرفت نيز با انتقاد از اعلام علنی اتهاماتی نظير اقدام عليه امنيت ملی و ارتباط با گروه‌های برانداز و معاند و ... که از قول مقامات قضايی در رسانه‌های حکومتی به اين جوانان نسبت داده می‌شود می‌گويد: "به موجب قانون تا کسی محاکمه و محکوم نشود حق ندارند علناً اتهام او را اعلام کنند. در غير اين صورت اعلام کننده می‌تواند به عنوان مفتری تحت تعقيب قرار گيرد. اين قانونی است که همه بايستی به آن پای‌بند باشيم اما متأسفانه گاه ديده می‌شود مقاماتی که خود مسئول اجرای قانون هستند به آن پای‌بند نيستند. روناک و هانا را در حالی متهم می‌کنند که پرونده هنوز در مرحله بازجويی است، به دادگاه نرفته و اتهام ثابت نشده است و من بر اين باور نيستم که اين دو دختر جوان امنيت ملی را به مخاطره انداخته باشند، آنان گناهی جز برابری‌خواهی ندارند." (گفت‌و‌گو با سايت تغير برای برابری)

آری همان گونه که شيرين عبادی در بخش ديگری از همين گفت‌و‌گو با اشاره به اتهام اقدام عليه امنيت ملی و تشويش اذهان عمومی که ديگر فعالان جنبش زنان هم به خاطر آن محاکمه و محکوم می‌گردند می‌گويد: "متأسفانه مدتی است می‌بينيم هر امر کوچک و جزئی را منتسب به امنيت ملی می‌کنند." هر آن‌کس که خبرهای مربوط به دستگيری، بازداشت و صدور احکام برای فعالان مدنی در ايران را حتی به صورت تيتروار دنبال کرده باشد می‌داند اتهام "اقدام عليه امنيت ملی" اتهام مشترک اکثر فعالان جنبش‌های اجتماعی و سياسی‌‌ست که از سوی حاکميت به آنان نسبت داده می‌شود و از سوی ديگر آنان که اندک شناختی از وضعيت اجتماعی و سياسی کردستان دارند و از حضور و فعاليت احزاب اپوزسيون کُرد در منطقه که غالباً مشی مبارزه مسلحانه را نيز کنار گذاشته‌ و به فعاليت‌های سياسی روی آورده‌اند آگاهند می‌دانند همين حضور باعث گشته تا حکومت هرگونه فعاليت مدنی در منطقه را منتسب به اين احزاب بداند و برای برخورد با فعالان اجتماعی و سياسی کُرد از انگ تجزيه‌طلب و تروريست و جاسوس و ... استفاده نمايد تا ضمن توجيه فضای امنيتی حاکم بر کردستان، چهره‌ای غيرواقعی و خشنوت‌گرايانه از مبارزات برحق ملت کُرد که در پيوند با تلاش‌های ديگر هم‌ميهنان‌شان در ايران برای رسيدن به جامعه‌ای عاری از نابرابری‌ها و تبعيضات جنسيتی، ملی و ... است نشان دهد و زمينه‌ی فعاليت مشترک بين جنبش‌های اجتماعی در مناطق مختلف کشور را کم‌رنگ نمايد.

آن‌چه که در هفته‌های اخير در رابطه با فعاليت‌های روناک و هانا و ياسر از سوی رسانه‌های حکومتی در بوق و کرنا دميده شد چيزی جز ادامه‌ی همان سياست‌های خصمانه نيست و البته اين‌بار اهداف ديگری نيز مورد توجه است و آن اين‌که از يک سو با معرفی اين جوانان به عنوان افرادی خشونت‌طلب و تروريست ضمن کاستن از حمايت فعالان و نهادهای حقوق بشری از آنان از گسترش روزافزون کمپين "يک ميليون امضاء برای تغيير قوانين تبعيض‌آميز" نيز در مناطق کردنشين کشور که با تلاش کنشگران کُرد کمپين توانسته جايگاه نسبتاً مناسبی را در بين بسياری از مردم و به ويژه زنان ساکن شهرها و حتی روستاهای کردستان پيدا کند جلوگيری کنند و از سوی ديگر با اتهاماتی نظير دروغ‌گويی و فرصت‌طلبی و سوءاستفاده از کمپين برای پيشبرد مقاصد سياسی که با هيچ منطق و معياری سازگار نيست ـ از آن روی که طرح‌ و هدف کمپين که درخواست جهت تغيير ده مورد مشخص از قوانين تبعيض‌آميز عليه زنان و جمع‌آوری يک ميليون امضاء در تائيد اين خواسته و نهايتاً تسليم اين درخواست به نهاد قانون‌گذاری کشور يعنی مجلس شورای اسلامی می‌باشد مشخص است و جای هيچ‌گونه شک و شبهه‌ای در اين بين باقی نيست ـ کنشگران کمپين که با وجود تفاوت‌های فکری و عقيدتی توانسته‌اند با توافق بر سر يک سری از اصول و موارد مشخص شده در يک کارزار مشترک با يکديگر همراه شوند را نسبت به هم بدبين کرده و زمينه‌ی بی‌اعتمادی و در نتيجه انشقاق در صفوف اين حرکت سترگ را که می‌رود تا ضمن آگاه ساختن زنان اين سرزمين از حقوق خويش قوانين ناعادلانه را در جهت رسيدن به برابری تغيير دهد به وجود آورند.

نگين شيخ‌الاسلامی نيز با تائيد اين مطلب می‌افزايد: «دستگيری روناک و هانا و به دنبال آن اتهامات عنوان شده عليه آن‌ها لطمه‌ای بسيار بزرگ به انجمن زنان آذرمهر زد و حاشيه‌ای امنيتی برای ما ايجاد کرد به گونه‌ای که باور و اعتماد گروهی از خانواده‌ها که فرزندان‌شان در انجمن فعال بودند نسبت به فعاليت‌های مشخص و تعريف شده‌ی ما سلب شد از آن روی که می‌بينند مرتب گفته می‌شود دو تن از اعضای انجمن زنان آذرمهر تروريست هستند! و اين در مورد فعاليت کمپين در کردستان نيز صدق می‌کند.»

دبير انجمن زنان آذرمهر ادامه می‌دهد: «متأسفانه پس از اعلام اين اتهامات و تبليغات رسانه‌ای که در اين رابطه صورت گرفت گويا نهادهای حقوق بشری و فعالان مدنی هم باورشان شده که اين جوانان تروريست هستند زيرا ديگر هيچ صدايی در حمايت از روناک و هانا و ياسر از سازمان‌ها و افراد برنمی‌خيزد مگر اندک دوستان ما در کمپين يک ميليون امضاء که اميدواريم حمايت‌های‌شان همچنان ادامه يابد.»

پرواضح است مردمان کردستان و به‌ويژه زنان کُرد تحت حاکميت قوانين نابرابر جنسيتی، ملی و ... از ستمی مضاعف رنج می‌برند و طبيعی‌ست آنان که دغدغه‌ی دفاع از حقوق بشر و رسيدن به جامعه‌ی مدنی را دارند نمی‌توانند به موارد ديگر تبعيض‌های قانونی و نقض حقوق انسانی افراد بشر بی‌تفاوت باشند. يکی از نمونه‌های بارز اين نوع تبعيضات در کشور ما محروم بودن کودکان وابسته به اقليت‌های ملی از حق آموزش به زبان مادری‌ست که بخش ديگری از فعاليت‌های روناک و هانا و ياسر مربوط می‌شود به تلاش‌های‌شان در جهت رفع ستم ملی و از آن ميان بحث مربوط به زبان مادری (کُردی) که نگين شيخ‌الاسلامی در پايان بيش‌تر راجع‌به اين موضوع و چگونگی فعاليت روناک و هانا در اين زمينه توضيح می‌دهد: «سخن گفتن و آموزش ديدن به زبان مادری از جمله حقوق ابتدايی و انسانی هر شهروند است و در اعلاميه جهانی حقوق بشر نيز به آن اشاره شده. همچنين حق آموزش و تدريس زبان‌های غيرفارسی از جمله زبان کُردی در مدارس به صراحت در اصل ۱۵ قانون اساسی جمهوری اسلامی ايران آمده و ما هم اين را حق مسلم خود می‌دانيم که بخواهيم به زبان خود بگوئيم و بخوانيم و بنويسم زيرا اين زبان ماست که به ملت‌مان هويت می‌بخشد و خواندن و نوشتن به آن حس حقارت و کوچک بودن را از فرزندان‌مان دور می‌کند و در رشد شخصيتی و افزايش اعتماد به نفس در آنان بسيار تأثيرگذار است. روناک و هانا هم با توجه به اين مهم با برگزاری کلاس‌هايی در شهر و روستاها فعاليت‌هايی را در زمينه‌ی آموزش زبان کُردی به صورت علمی به مادران برای انتقال درست‌تر و بهتر به فرزندان‌شان انجام می‌دادند و اعتقادشان بر اين بود که زنان کُرد بايد با افزايش آگاهی و توان‌مندی خود آموزش مناسبی به فرزندان‌شان بدهند و نسلی را تربيت کنند تا ضمن پاسداری از فرهنگ و زبان‌شان در رشد و پيشرفت سرزمين‌شان نقشی مثبت و سازنده داشته باشند.»

*شوربختانه هم‌زمان با انتشار اين نوشتار باخبر شديم خانم فاطمه گفتاری (مادر ياسر گلی) پس از مراجعه به ستاد خبری اطلاعات شهر سنندج برای دريافت وسايل فرزند دربندش به دستور بازپرسی شعبه ۴ دادگاه انقلاب دستگير و روانه‌ی بازداشت‌گاه شده‌اند که مشخص است اين اقدام به واسطه‌ی پی‌گيری‌هايی بوده که اين مادر برای رهايی فرزند خويش داشته‌اند.
